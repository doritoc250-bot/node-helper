import express from 'express'; import { createServer } from 'http'; import { WebSocketServer } from 'ws'; import { fileURLToPath } from 'url'; import { join, dirname } from 'path'; import { existsSync } from 'fs'; const __dirname = dirname(fileURLToPath(import.meta.url)); console.log('dirname:', __dirname); console.log('public exists:', existsSync(join(__dirname, 'public'))); console.log('index exists:', existsSync(join(__dirname, 'public/index.html'))); const app = express(); const server = createServer(app); app.use('/uv/', express.static(join(__dirname, 'public/uv'))); app.use(express.static(join(__dirname, 'public'))); app.get('*', (req, res) => { const indexPath = join(__dirname, 'public/index.html'); if (existsSync(indexPath)) { res.sendFile(indexPath); } else { res.send('index.html not found at: ' + indexPath); } }); const wss = new WebSocketServer({ server, path: '/wisp/' }); wss.on('connection', (ws) => ws.on('error', console.error)); const port = parseInt(process.env.PORT || '8080'); server.listen(port, '0.0.0.0', () => { console.log(`Running on port ${port}`); setInterval(async () => { try { await fetch(`http://localhost:${port}`); } catch(e) {} }, 4 * 60 * 1000); });
